461 uint64_t OrderBook::CalculateTrailingStopPrice(const Order& order) const noexcept
462 {
463  // Get the current market price
464  uint64_t market_price = order.IsBuy() ? GetMarketStopPriceAsk() : GetMarketStopPriceBid();
465  int64_t trailing_distance = order.TrailingDistance;
466  int64_t trailing_step = order.TrailingStep;
467 
468  // Convert percentage trailing values into absolute ones
469  if (trailing_distance < 0)
470  {
471  trailing_distance = (int64_t)((-trailing_distance * market_price) / 10000);
472  trailing_step = (int64_t)((-trailing_step * market_price) / 10000);
473  }
474 
475  uint64_t old_price = order.StopPrice;
476 
477  if (order.IsBuy())
478  {
479  // Calculate a new stop price
480  uint64_t new_price = (market_price < (std::numeric_limits<uint64_t>::max() - trailing_distance)) ? (market_price + trailing_distance) : std::numeric_limits<uint64_t>::max();
481 
482  // If the new price is better and we get through the trailing step
483  if (new_price < old_price)
484  if ((old_price - new_price) >= (uint64_t)trailing_step)
485  return new_price;
486  }
487  else
488  {
489  // Calculate a new stop price
490  uint64_t new_price = (market_price > (uint64_t)trailing_distance) ? (market_price - trailing_distance) : 0;
491 
492  // If the new price is better and we get through the trailing step
493  if (new_price > old_price)
494  if ((new_price - old_price) >= (uint64_t)trailing_step)
495  return new_price;
496  }
497 
498  return old_price;
499 }
